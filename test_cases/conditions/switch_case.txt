int main() {
    int x = 2;
    switch (x) {
        case 1:
            print("x is 1");
            break;
        case 2:
            print("x is 2");
            break;
        case 3:
            print("x is 3");
            break;
        default:
            print("x is not 1, 2, or 3");
            break;
    }
    return 0;
}
