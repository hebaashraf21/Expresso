int main() {
    int a = 5;
    int b = 10;

    /* Mathematical expressions */
    int neg = -a;

    int pre_incr = ++a;

    int post_incr = b++;

    int pre_dec = --a;

    int post_dec = b--;

    int add = a + b;

    int sub = a - b;

    int mul = a * b;

    int div = a / b;

    int pow = a ^ b;

    int mod = a % b;


    /* Logical expressions */
    bool and_expr = (a < b) && (a > 0);

    bool or_expr = (a > b) || (b > 0);

    bool not_expr = !and_expr;


    /* Comparison expressions */
    bool eq = (a == b);

    bool neq = (a != b);

    bool lt = (a < b);

    bool ltq = (a <= b);

    bool gt = (a > b);

    bool gtq = (a >= b);


    return 0;
}
