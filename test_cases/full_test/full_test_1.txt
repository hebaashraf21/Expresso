int add(int x, int y) {
    return x + y;
}

int main() {
    // Variable declarations and assignments
    int a = 10;
    float b = 3.14;
    char c = 'A';
    bool d = true;
    string e = "Hello, world!";

    // Print statements
    print("a: " + a);
    print("b: " + b);
    print("c: " + c);
    print("d: " + d);
    print("e: " + e);

    // Mathematical expressions
    int sum = add(a, 20);
    int diff = a - 5;
    int prod = a * 2;
    float quot = b / 2.0;
    int power = a ^ 2;
    int mod = a % 3;

    // Logical expressions
    if (a < 20 && d) {
        print("a is less than 20 and d is true");
    }

    // Comparison expressions
    if (a == 10) {
        print("a is equal to 10");
    } else {
        print("a is not equal to 10");
    }

    // Increment and decrement operations
    a++;
    a--;


    // Loops with break and continue
    for (int i = 0; i < 10; i++) {
        print( i);
    }

    int j = 0;
    while (j < 10) {
        if (j == 2) {
            j++;
            continue; // Skip the rest of the loop when j is 2
        }
        if (j == 5) {
            break; // Exit the loop when j is 5
        }
        print(j);
        j++;
    }

    int k = 0;
    repeat {
        if (k == 1) {
        print(k);
        k++;
        }
    } until (k >= 5);

    // Switch-case statement
    int day = 3;
    switch (day) {
        case 1:
            print("Monday");
            break;
        case 2:
            print("Tuesday");
            break;
        case 3:
            print("Wednesday");
            break;
        case 4:
            print("Thursday");
            break;
        case 5:
            print("Friday");
            break;
        default:
            print("Weekend");
            break;
    }

    return 0;
}
